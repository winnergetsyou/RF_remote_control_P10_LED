//----------------------------------------------
void setup()
{
  Timer1.initialize(2000);
  Timer1.attachInterrupt(Scan);
  dmd.clearScreen(true); // clears your LED screen
  Serial.begin(9600);
  pinMode(2, INPUT);
  pinMode(3, INPUT);
  pinMode(4, INPUT);
  pinMode(5, INPUT);
  attachInterrupt(digitalPinToInterrupt(2), add, FALLING);
  attachInterrupt(digitalPinToInterrupt(3), sub, FALLING);
}
